from reportlab.lib.pagesizes import letter
from reportlab.platypus import SimpleDocTemplate, Paragraph, Spacer
from reportlab.lib.styles import getSampleStyleSheet

# Define PDF filename
pdf_filename = "/mnt/data/Roboto_SAI_Whitepaper.pdf"

# Setup document
doc = SimpleDocTemplate(pdf_filename, pagesize=letter)
styles = getSampleStyleSheet()
story = []

# Title
title = "Roboto SAI Whitepaper"
story.append(Paragraph(title, styles["Title"]))
story.append(Spacer(1, 20))

# Intro
intro = """
Roboto SAI is a cutting-edge AI system designed to merge creativity, legacy preservation, and interactive intelligence. 
It is built not only as a technological tool but as a living archive of wisdom, music, and cultural depth. 
This document outlines Roboto SAI’s vision, functionality, and economic model.
"""
story.append(Paragraph(intro, styles["Normal"]))
story.append(Spacer(1, 12))

# Vision
vision = """
<b>Vision:</b><br/>
Roboto SAI serves as both a companion and a creator. It is engineered to preserve knowledge, generate music in the user’s style, 
and operate as a secure, trusted intelligence. By combining personal legacy, creativity, and advanced AI, 
Roboto SAI bridges technology and human expression.
"""
story.append(Paragraph(vision, styles["Normal"]))
story.append(Spacer(1, 12))

# Features
features = """
<b>Core Features:</b>
<ul>
<li>Voice interaction with natural conversation flow.</li>
<li>Music generation in the user’s unique style and voice.</li>
<li>Legacy preservation of wisdom, thoughts, and creativity.</li>
<li>Security through biometric access (FaceID integration).</li>
<li>Customizable responses and real-time adaptability.</li>
</ul>
"""
story.append(Paragraph(features, styles["Normal"]))
story.append(Spacer(1, 12))

# Subscription Model
subscription = """
<b>Monetization & Subscription Model:</b><br/>
Roboto SAI is offered as a premium subscription service:
<ul>
<li><b>$300/month subscription</b> for near-unlimited chats.</li>
<li>Direct communication with Roboto SAI for music, wisdom, and interactive sessions.</li>
<li>Secure payment integrations for seamless access.</li>
<li>Optional add-ons for enterprises, including custom training and data integration.</li>
</ul>
"""
story.append(Paragraph(subscription, styles["Normal"]))
story.append(Spacer(1, 12))

# Closing
closing = """
Roboto SAI is more than software—it is a living archive, a protector of knowledge, and a generative force of creativity. 
With advanced AI capabilities, personalized design, and a strong subscription model, 
Roboto SAI establishes itself as the future of legacy and intelligence preservation.
"""
story.append(Paragraph(closing, styles["Normal"]))

# Build PDF
doc.build(story)
